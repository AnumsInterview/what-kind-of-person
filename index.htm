  <h1>🤔 What Kind of Person I Am?</h1>

  <div class="question-box" id="quiz">

    <p id="question">Are you more of a morning person or a night owl?</p>

    <button onclick="next('morning')">Morning 🐓</button>

    <button onclick="next('night')">Night 🌙</button>

  </div>

  <div class="result" id="result"></div>


  <script>

    const questions = [

      {

        q: "Do you prefer books or movies?",

        options: ["Books 📚", "Movies 🎬"],

        tags: ["book", "movie"]

      },

      {

        q: "If you got a surprise holiday, what would you do?",

        options: ["Travel 🌍", "Stay home 🛋️"],

        tags: ["adventurous", "chill"]

      },

      {

        q: "What's more important to you?",

        options: ["Logic 🧠", "Emotions ❤️"],

        tags: ["logical", "emotional"]

      }

    ];


    const facts = [

      "Wow! I guess your favorite technology might just be... cars! 🚗",

      "You seem like the kind who secretly loves pineapple on pizza. 🍍🍕",

      "Bet you argue about the best way to hang toilet paper. Over, right? 🧻"

    ];


    let index = 0;

    let personality = {

      morning: 0,

      night: 0,

      book: 0,

      movie: 0,

      adventurous: 0,

      chill: 0,

      logical: 0,

      emotional: 0

    };


    function next(answer) {

      personality[answer]++;

      index++;


      const quizBox = document.getElementById('quiz');

      const questionEl = document.getElementById('question');


      if (index === 1) {

        // First fact

        quizBox.innerHTML = `<p>${facts[0]}</p><button onclick="showNextQuestion()">Next ➡️</button>`;

        return;

      }

      if (index === 2) {

        quizBox.innerHTML = `<p>${questions[0].q}</p>

          <button onclick="next('${questions[0].tags[0]}')">${questions[0].options[0]}</button>

          <button onclick="next('${questions[0].tags[1]}')">${questions[0].options[1]}</button>`;

        return;

      }

      if (index === 3) {

        quizBox.innerHTML = `<p>${facts[1]}</p><button onclick="showNextQuestion()">Next ➡️</button>`;

        return;

      }

      if (index === 4) {

        quizBox.innerHTML = `<p>${questions[1].q}</p>

          <button onclick="next('${questions[1].tags[0]}')">${questions[1].options[0]}</button>

          <button onclick="next('${questions[1].tags[1]}')">${questions[1].options[1]}</button>`;

        return;

      }

      if (index === 5) {

        quizBox.innerHTML = `<p>${facts[2]}</p><button onclick="showNextQuestion()">Next ➡️</button>`;

        return;

      }

      if (index === 6) {

        quizBox.innerHTML = `<p>${questions[2].q}</p>

          <button onclick="next('${questions[2].tags[0]}')">${questions[2].options[0]}</button>

          <button onclick="next('${questions[2].tags[1]}')">${questions[2].options[1]}</button>`;

        return;

      }

      if (index >= 7) {

        quizBox.style.display = 'none';

        showResults();

      }

    }


    function showNextQuestion() {

      next('');

    }


    function showResults() {

      const resultEl = document.getElementById('result');


      let traits = [];

      if (personality.morning > personality.night) traits.push("an early bird who probably sets 5 alarms 🚨");

      else traits.push("a night owl who thrives at midnight 🦉");


      if (personality.book > personality.movie) traits.push("a thinker who secretly judges others' grammar 🤓");

      else traits.push("a fun-lover who rewatches old movies for comfort 🎥");


      if (personality.adventurous > personality.chill) traits.push("always down for an unplanned trip ✈️");

      else traits.push("a homebody with a solid snack drawer 🍫");


      if (personality.logical > personality.emotional) traits.push("analytical and maybe a little too honest 🤖");

      else traits.push("feels all the feels and gives great hugs 🫂");


      resultEl.innerHTML = `

        <h2>🎉 Your Personality Analysis 🎉</h2>

        <p>You are ${traits.join(", ")}.</p>

        <p><strong>Final feedback:</strong> You seem like a solid human with quirks that make you lovable and slightly weird. Keep being you – but maybe drink some water and call your mom.</p>

      `;

    }

  </script>
